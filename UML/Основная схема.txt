    (Интерфейс)
    [Локальное хранилище]
    [Контроллер фронтэнда]
}
package "Бэкэнд" {
	[Роутер]
	[Представление]
    [Контроллер]
   	node "Парсер"
	package "Модель" {
    	database "База данных"
        [Управляющий код модели]
	}
}
(HTTP) ..> [Роутер]
(Ajax) ..> [Роутер]
(Ajax) <-- [Управляющий код модели]
[Пользователь] ..> (Интерфейс)
[Пользователь] ..> (HTTP)
[Контроллер фронтэнда] <-> (Ajax)
[Контроллер фронтэнда] -- (Интерфейс)
[Контроллер фронтэнда] <--> [Локальное хранилище]
[Контроллер фронтэнда] -> [Отображение]
[Представление] --> (HTTP)
[Представление] <-- [Управляющий код модели]
[Контроллер] ..> [Управляющий код модели]
(HTTP) --> [Отображение]
[Роутер] --> [Контроллер]
[Управляющий код модели] <--> [База данных]
[Управляющий код модели] <-- [Парсер]
[Управляющий код модели] ..> [Парсер]
[Отображение] --> [Пользователь]
note left of HTTP : Для прямых заходов и\nпереходов между страницами.
